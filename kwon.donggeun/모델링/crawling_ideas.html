# 네이버 영화 검색 크롤링 예시
import requests
from bs4 import BeautifulSoup

def get_naver_rating(title):
    url = f"https://movie.naver.com/movie/search/result.nhn?query={title}§ion=movie"
    res = requests.get(url, headers={"User-Agent": "Mozilla/5.0"})
    soup = BeautifulSoup(res.text, "html.parser")
    rating = soup.select_one(".star_score em")
    return float(rating.text) if rating else None

🆕 blockbuster_ratio

관객 100만 이상 영화 시청 비율. 메인스트림 콘텐츠 선호도.

blockbuster_ratio indie_ratio
# KOBIS 데이터 조인 (바로 실행 가능)
kobis = pd.read_csv("data/Movie_Master_kobis.csv")
vh = pd.read_csv("data/View_History.csv")

# View_History × KOBIS 조인
vh_kobis = vh.merge(kobis[["MOVIE_ID", "audiCnt"]], on="MOVIE_ID", how="left")

# 유저별 평균 관객수 집계
user_stats = vh_kobis.groupby("USER_ID")["audiCnt"].agg(
    avg_audience_cnt="mean",
    blockbuster_ratio=lambda x: (x > 1_000_000).mean()
).reset_index()

# Membership에 조인
df = df.merge(user_stats, on="USER_ID", how="left")
